// Fonction pour charger et afficher les messages function chargerMessages() { fetch('backend/get_messages.php') .then(response => response.json()) .then(messages => { const tableBody = document.querySelector('#tableau-messages tbody'); tableBody.innerHTML = ''; // On vide le tableau avant d'insérer // Boucle sur chaque message reçu du PHP messages.forEach(msg => { const ligne = ` ${msg.date} ${msg.nom} ${msg.sujet} `; tableBody.insertAdjacentHTML('beforeend', ligne); }); }) .catch(error => console.error('Erreur au chargement :', error)); } // Appeler la fonction au chargement de la page admin document.addEventListener('DOMContentLoaded', chargerMessages);
